# split

  • return the list of string which are splited.
a = "   Hello 1 2 3     5    6 7    9 10 0"
print(a.split())

b = "Hello,1,2,3,5,9"
print(b.split(","))

c = "1-2-3-4-5-6    -7"
print(c.split("-"))
