Solution 
k = int(input("Enter k: "))

spacebar_a0 = k - 1
spacebar_d  = -1

star_a0 = 1
star_d = 2

for i in range(k):
	for s in range( spacebar_a0 + spacebar_d*i ):
		print(" ", end="")
	for j in range( star_a0 +  star_d*i ):
		print("*", end="")
	print()
