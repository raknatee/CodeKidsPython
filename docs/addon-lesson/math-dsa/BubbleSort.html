# Python Implementation

arr = [564, 11, 300, 0, -90, 60, 60]

def bubble_sort(arr):
    for i in range(0,len(arr)):
        for j in range(0,len(arr)-1):
            if(arr[j]>arr[j+1]):
                arr[j],arr[j+1] = arr[j+1],arr[j]

bubble_sort(arr)
